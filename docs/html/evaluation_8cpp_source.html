37 void evaluationCPU(Individual *const subpop, const int nIndividuals, const float *const trDataBase, const int *const selInstances, const int nThreads, const Config *const conf) {

40  /************ K-means algorithm in C++ ***********/
41 
42  #pragma omp parallel num_threads(nThreads) if (nThreads > 1)
43  {
44  const int totalCoord = conf -> K * conf -> nFeatures;
45  unsigned char mapping[conf -> trNInstances];
46  float centroids[totalCoord];
47  float distCentroids[conf -> trNInstances];
48  int samples_in_k[conf -> K];
49 
50  // Evaluate all individuals
51  #pragma omp for
52  for (int ind = 0; ind < nIndividuals; ++ind) {
53 
54  // The centroids will have the selected features of the individual
55  for (int k = 0; k < conf -> K; ++k) {
56  int posTrDataBase = selInstances[k] * conf -> nFeatures;
57  int posCentr = k * conf -> nFeatures;
58 
59  for (int f = 0; f < conf -> nFeatures; ++f) {
60  centroids[posCentr + f] = trDataBase[posTrDataBase + f];
61  }
62  }
63 
64  // Initialize the mapping table
65  for (int i = 0; i < conf -> trNInstances; ++i) {
66  mapping[i] = 0;
67  }
68 
69 
70  /******************** Convergence process *********************/
71 
72  // To avoid poor performance, 'conf -> maxIterKmeans' iterations are executed
73  for (int maxIter = 0; maxIter < conf -> maxIterKmeans; ++maxIter) {
74 
75  for (int k = 0; k < conf -> K; ++k) {
76  samples_in_k[k] = 0;
77  }
78 
79  // Calculate all distances (Euclidean distance) between each instance and the centroids
80  for (int i = 0; i < conf -> trNInstances; ++i) {
81  float minDist = INFINITY;
82  int selectCentroid;
83  int pos = conf -> nFeatures * i;
84  for (int k = 0, posCentr = 0; k < conf -> K; ++k, posCentr += conf -> nFeatures) {
85  float dist = 0.0f;
86  for (int f = 0; f < conf -> nFeatures; ++f) {
87  if (subpop[ind].chromosome[f]) {
88  float dif = trDataBase[pos + f] - centroids[posCentr + f];
89  dist += dif * dif;
90  }
91  }
92 
93  if (dist < minDist) {
94  minDist = dist;
95  selectCentroid = k;
96  }
97  }
98 
99  distCentroids[i] = minDist;
100  samples_in_k[selectCentroid]++;
101  if (mapping[i] != selectCentroid) {
102  mapping[i] = selectCentroid;
103  }
104  }
105 
106  // Update the position of the centroids
107  for (int f = 0; f < conf -> nFeatures; ++f) {
108  if (subpop[ind].chromosome[f]) {
109  for (int k = 0; k < conf -> K; ++k) {
110  float sum = 0.0f;
111  for (int i = 0; i < conf -> trNInstances; ++i) {
112  if (mapping[i] == k) {
113  sum += trDataBase[(conf -> nFeatures * i) + f];
114  }
115  }
116  centroids[(k * conf -> nFeatures) + f] = (samples_in_k[k] > 0) ? sum / samples_in_k[k] : centroids[(k * conf -> nFeatures) + f];
117  }
118  }
119  }
120  }
121 
122 
123  /************ Minimize the within-cluster and maximize Inter-cluster sum of squares (WCSS and ICSS) *************/
124 
125  float sumWithin = 0.0f;
126  float sumInter = 0.0f;
127 
128  // Within-cluster
129  for (int i = 0; i < conf -> trNInstances; ++i) {
130  sumWithin += sqrt(distCentroids[i]);
131  }
132 
133  // Inter-cluster
134  for (int posCentr = 0; posCentr < totalCoord; posCentr += conf -> nFeatures) {
135  for (int i = posCentr + conf -> nFeatures; i < totalCoord; i += conf -> nFeatures) {
136  float sum = 0.0f;
137  for (int f = 0; f < conf -> nFeatures; ++f) {
138  if (subpop[ind].chromosome[f]) {
139  sum += (centroids[posCentr + f] - centroids[i + f]) * (centroids[posCentr + f] - centroids[i + f]);
140  }
141  }
142  sumInter += sqrt(sum);
143  }
144  }
145 
146  // First objective function (Within-cluster sum of squares (WCSS))
147  subpop[ind].fitness[0] = sumWithin;
148 
149  // Second objective function (Inter-cluster sum of squares (ICSS))
150  subpop[ind].fitness[1] = sumInter;
151  }
152  }
153 }

240 void normalizeFitness(Individual *const subpop, const int nIndividuals, const Config *const conf) {
241 
242  for (unsigned char obj = 0; obj < conf -> nObjectives; ++obj) {
243 
244  // Fitness vector average
245  float average = 0;
246  for (int i = 0; i < nIndividuals; ++i) {
247  average += subpop[i].fitness[obj];
248  }
249 
250  average /= nIndividuals;
251 
252  // Fitness vector variance
253  float variance = 0;
254  for (int i = 0; i < nIndividuals; ++i) {
255  variance += (subpop[i].fitness[obj] - average) * (subpop[i].fitness[obj] - average);
256  }
257  variance /= (nIndividuals - 1);
258 
259  // Fitness vector standard deviation
260  float std_deviation = sqrt(variance);
261 
262  // The second objective is a maximization problem. x_new must be negative
263  if (obj == 1) {
264 
265  // Normalize a set of continuous values using SoftMax (based on the logistic function)
266  for (int i = 0; i < nIndividuals; ++i) {
267  float x_scaled = (subpop[i].fitness[obj] - average) / std_deviation;
268  float x_new = 1.0f / (1.0f + exp(-x_scaled));
269  subpop[i].fitness[obj] = -x_new;
270  }
271  }
272  else {
273 
274  // Normalize a set of continuous values using SoftMax (based on the logistic function)
275  for (int i = 0; i < nIndividuals; ++i) {
276  float x_scaled = (subpop[i].fitness[obj] - average) / std_deviation;
277  subpop[i].fitness[obj] = 1.0f / (1.0f + exp(-x_scaled));
278  }
279  }
