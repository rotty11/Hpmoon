159bool CmdParser::parse(const char *const *const argv, const int argc) {
160
161 bool success = (!this -> arguments.empty() && argv != NULL && argc > 0);
162 int parsed = 0;
163 for(auto it = this -> arguments.begin(); it != this -> arguments.end() && success; ++it) {
164
165 // Delete a possible previous parse;
166 it -> second.setValue("", true);
167 for (int i = 0; i < argc && argv[i] != NULL && success; ++i) {
168
169 // Check if this argument has been passed in the command line string and needs a value
170 if (it -> first.compare(argv[i]) == 0) {
171 if (it -> second.isValueRequired()) {
172 success = (i == argc - 1 || this -> find(argv[i + 1]) || strcmp(argv[i + 1], "") == 0) ? false : it -> second.setValue(argv[i + 1]);
173 }
174 else {
175 success = it -> second.setValue("");
176 }
177 ++parsed;
178 }
179 }
180 }
181
182 return (success && parsed > 0);
183}

211void CmdParser::printHelp() {
212
213 std::string out;
214 if (!this -> overview.empty()) {
215 out += '\n' + this -> overview + '\n';
216 }
217 if (!this -> syntax.empty()) {
218 out += "\nSYNTAX: " + this -> syntax + '\n';
219 }
220 if (!this -> arguments.empty()) {
221 out += "\nARGS:\n";
222
223 // Find the longest argument and save its length
224 int maxLength = 0;
225 for(const auto &it : this -> arguments) {
226 maxLength = std::max((int) it.first.length(), maxLength);
227 }
228
229 // Format the arguments
230 for(auto &it : this -> arguments) {
231 int blankSpaces = maxLength - it.first.length() + 3;
232 out += "\n" + it.first;
233 if (it.second.isValueRequired()) {
234 out += " VAL";
235 }
236 else {
237 blankSpaces += 4;
238 }
239 for (int i = 0; i < blankSpaces; ++i) {
240 out += ' ';
241 }
242 out += it.second.getDescription() + '\n';
243 }
244
245 if (!this -> examples.empty()) {
246 out += "\nEXAMPLES:\n\n";
247 for (const auto &it : this -> examples) {
248 out += it + '\n';
249 }
250 }
251 }
252
253 if (!this -> footer.empty()) {
254 out += '\n' + this -> footer + '\n';
255 }
256 if (!out.empty()) {
257 out += '\n';
258 fprintf(stdout, "%s", out.c_str());
259 }
260}
