31void normDataBase(float *const dataBase, const Config *const conf) {
32
33
34 /********** Database normalization ***********/
35
36 for(int j = 0; j < conf -> nFeatures; ++j) {
37
38 // Average of the features vector
39 float average = 0;
40 for(int i = 0; i < conf -> trNInstances; ++i) {
41 int pos = (conf -> nFeatures * i) + j;
42 average += dataBase[pos];
43 }
44
45 average /= conf -> trNInstances;
46
47 // Variance of the features vector
48 float variance = 0;
49 for(int i = 0; i < conf -> trNInstances; ++i) {
50 int pos = (conf -> nFeatures * i) + j;
51 variance += (dataBase[pos] - average) * (dataBase[pos] - average);
52 }
53 variance /= (conf -> trNInstances - 1);
54
55 // Standard deviation of the features vector
56 float std_deviation = sqrt(variance);
57
58 // Normalize a set of continuous values using SoftMax (based on the logistic function)
59 for(int i = 0; i < conf -> trNInstances; ++i) {
60 int pos = (conf -> nFeatures * i) + j;
61 float x_scaled = (dataBase[pos] - average) / std_deviation;
62 dataBase[pos] = 1.0f / (1.0f + exp(-x_scaled));
63 }
64 }
65}

83

94 ++nCols;
95 }

98 ++nRows;
99 std::stringstream aux(line);
100 int tmp = 0;
101 while(aux >> dato) {
102 ++tmp;
103 }
104 check(tmp != nCols, "%s %d\n", BD_ERROR_ROW_UNEQUAL, nRows);
105 }
106
107
108 /********** Check the parameters specified in configuration ***********/
109
110 check(nRows < 4 || nCols < 4, "%s\n", BD_ERROR_DIMENSIONS_MIN);
111 check(conf -> trNInstances < 4 || conf -> trNInstances > nRows, "%s %d\n", BD_ERROR_INSTANCES_RANGE, nRows);
112 check(conf -> nFeatures != nCols, "%s\n", BD_ERROR_COLUMNS_UNEQUAL);
113
114
115 /********** Reading and database storage ***********/
116
117 fData.clear();
118 fData.seekg(0);
119 int dbSize = conf -> trNInstances * nCols;
120 float *dataBase = new float[dbSize];
121 for (int i = 0; i < dbSize; ++i) {
122 fData >> dataBase[i];
123 }
124 fData.close();
125
126 // Normalize the database if it is required and return it
127 if (conf -> trNormalize) {
128 normDataBase(dataBase, conf);
129 }
130 return dataBase;
131}
