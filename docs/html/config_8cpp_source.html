300int split(const std::string str, std::string *&tokens) {
301
302 std::stringstream ss(str);
303 std::string token;
304 int nTokens = 0;
305 while(getline(ss, token, ',')) {
306 ++nTokens;
307 }
308
309 tokens = new std::string[nTokens];
310 if (nTokens > 0) {
311 std::stringstream aux(str);
312 for (int i = 0; i < nTokens; ++i) {
313 getline(aux, tokens[i], ',');
314 }
315 }
316
317 return nTokens;
318}
