52CLDevice *createDevices(const float *const trDataBase, const int *const selInstances, const float *const transposedTrDataBase, Config *const conf) {
53
54
55 /********** Find the OpenCL devices specified in configuration ***********/
56
57 // OpenCL variables
58 cl_uint numPlatformsDevices;
59 cl_device_type deviceType;
60 cl_program program;
61 cl_kernel kernel;
62 cl_int status;
63
64 // Others variables
65 auto allDevices = getAllDevices();
66 CLDevice *devices = new CLDevice[conf -> nDevices + (conf -> ompThreads > 0)];
67
68 for (int dev = 0; dev < conf -> nDevices; ++dev) {
69
70 bool found = false;
71 for (int allDev = 0; allDev < allDevices.size() && !found; ++allDev) {
72
73 // Get the specified OpenCL device
74 char dbuff[120];
75 check(clGetDeviceInfo(allDevices[allDev], CL_DEVICE_NAME, sizeof(dbuff), dbuff, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_DEVICE_NAME);
76
77 // If the device exists...
78 if (conf -> devices[dev] == dbuff) {
79 devices[dev].device = allDevices[allDev];
80 devices[dev].deviceName = dbuff;
81 check(clGetDeviceInfo(devices[dev].device, CL_DEVICE_TYPE, sizeof(cl_device_type), &(devices[dev].deviceType), NULL) != CL_SUCCESS, "%s\n", CL_ERROR_DEVICE_TYPE);
82
83
84 /********** Device local memory usage ***********/
85
86 long int usedMemory = conf -> nFeatures * sizeof(cl_uchar); // Chromosome of the individual
87 usedMemory += conf -> trNInstances * sizeof(cl_uchar); // Mapping buffer
88 usedMemory += conf -> K * conf -> nFeatures * sizeof(cl_float); // Centroids buffer
89 usedMemory += conf -> trNInstances * sizeof(cl_float); // DistCentroids buffer
90 usedMemory += conf -> K * sizeof(cl_int); // Samples_in_k buffer
91
92 // Get the maximum local memory size
93 long int maxMemory;
94 check(clGetDeviceInfo(devices[dev].device, CL_DEVICE_LOCAL_MEM_SIZE, sizeof(long int), &maxMemory, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_DEVICE_MAXMEM);
95
96 // Avoid exceeding the maximum local memory available. 1024 bytes of margin
97 check(usedMemory > maxMemory - 1024, "%s:\n\tMax memory: %ld bytes\n\tAllow memory: %ld bytes\n\tUsed memory: %ld bytes\n", CL_ERROR_DEVICE_LOCALMEM, maxMemory, maxMemory - 1024, usedMemory);
98
99
100 /********** Create context ***********/
101
102 devices[dev].context = clCreateContext(NULL, 1, &(devices[dev].device), 0, 0, &status);
103 check(status != CL_SUCCESS, "%s\n", CL_ERROR_DEVICE_CONTEXT);
104
105
106 /********** Create command queue ***********/
107
108 devices[dev].commandQueue = clCreateCommandQueue(devices[dev].context, devices[dev].device, CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE | CL_QUEUE_PROFILING_ENABLE, &status);
109 check(status != CL_SUCCESS, "%s\n", CL_ERROR_DEVICE_QUEUE);
110
111
112 /********** Read the kernel file ***********/
113
114 // Open the file containing the kernels
115 std::fstream kernels(conf -> kernelsFileName.c_str(), std::fstream::in);
116 check(!kernels.is_open(), "%s\n", CL_ERROR_FILE_OPEN);
117
118 // Obtain the size
119 kernels.seekg(0, kernels.end);
120 size_t fSize = kernels.tellg();
121 kernels.seekg(0, kernels.beg);
122
123 char *kernelSource = new char[fSize];
124 kernels.read(kernelSource, fSize);
125 kernels.close();
126
127 // Create program
128 program = clCreateProgramWithSource(devices[dev].context, 1, (const char **) &kernelSource, &fSize, &status);
129 check(status != CL_SUCCESS, "%s\n", CL_ERROR_PROGRAM_BUILD);
130
131 // Build program for the device in the context
132 char buildOptions[196];
133 sprintf(buildOptions, "-I include -D N_INSTANCES=%d -D N_FEATURES=%d -D N_OBJECTIVES=%d -D K=%d -D MAX_ITER_KMEANS=%d", conf -> trNInstances, conf -> nFeatures, conf -> nObjectives, conf -> K, conf -> maxIterKmeans);
134 if (clBuildProgram(program, 1, &(devices[dev].device), buildOptions, 0, 0) != CL_SUCCESS) {
135 char buffer[4096];
136 fprintf(stderr, "Error: Could not build the program\n");
137 check(clGetProgramBuildInfo(program, devices[dev].device, CL_PROGRAM_BUILD_LOG, sizeof(buffer), buffer, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_PROGRAM_ERRORS);
138 check(true, "%s\n", buffer);
139 }
140
141 /********** Create kernel ***********/
142
143 const char *kernelName = (devices[dev].deviceType == CL_DEVICE_TYPE_GPU) ? "kmeansGPU" : "";
144 devices[dev].kernel = clCreateKernel(program, kernelName, &status);
145 check(status != CL_SUCCESS, "%s\n", CL_ERROR_KERNEL_BUILD);
146
147
148 /******* Work-items *******/
149
150 devices[dev].computeUnits = atoi(conf -> computeUnits[dev].c_str());
151 devices[dev].wiLocal = atoi(conf -> wiLocal[dev].c_str());
152 devices[dev].wiGlobal = devices[dev].computeUnits * devices[dev].wiLocal;
153
154
155 /******* Create and write the databases and centroids buffers. Create the subpopulations buffer. Set kernel arguments *******/
156
157 // Create buffers
158 devices[dev].objSubpopulations = clCreateBuffer(devices[dev].context, CL_MEM_READ_WRITE, conf -> familySize * sizeof(Individual), 0, &status);
159 check(status != CL_SUCCESS, "%s\n", CL_ERROR_OBJECT_SUBPOPS);
160
161 devices[dev].objTrDataBase = clCreateBuffer(devices[dev].context, CL_MEM_READ_ONLY, conf -> trNInstances * conf -> nFeatures * sizeof(cl_float), 0, &status);
162 check(status != CL_SUCCESS, "%s\n", CL_ERROR_OBJECT_TRDB);
163
164 devices[dev].objTransposedTrDataBase = clCreateBuffer(devices[dev].context, CL_MEM_READ_ONLY, conf -> trNInstances * conf -> nFeatures * sizeof(cl_float), 0, &status);
165 check(status != CL_SUCCESS, "%s\n", CL_ERROR_OBJECT_TTRDB);
166
167 devices[dev].objSelInstances = clCreateBuffer(devices[dev].context, CL_MEM_READ_ONLY, conf -> K * sizeof(cl_int), 0, &status);
168 check(status != CL_SUCCESS, "%s\n", CL_ERROR_OBJECT_CENTROIDS);
169
170 // Sets kernel arguments
171 check(clSetKernelArg(devices[dev].kernel, 0, sizeof(cl_mem), (void *)&(devices[dev].objSubpopulations)) != CL_SUCCESS, "%s\n", CL_ERROR_KERNEL_ARGUMENT1);
172
173 check(clSetKernelArg(devices[dev].kernel, 1, sizeof(cl_mem), (void *)&(devices[dev].objSelInstances)) != CL_SUCCESS, "%s\n", CL_ERROR_KERNEL_ARGUMENT2);
174
175 check(clSetKernelArg(devices[dev].kernel, 2, sizeof(cl_mem), (void *)&(devices[dev].objTrDataBase)) != CL_SUCCESS, "%s\n", CL_ERROR_KERNEL_ARGUMENT3);
176
177 check(clSetKernelArg(devices[dev].kernel, 5, sizeof(cl_mem), (void *)&(devices[dev].objTransposedTrDataBase)) != CL_SUCCESS, "%s\n", CL_ERROR_KERNEL_ARGUMENT6);
178
179 // Write buffers
180 check(clEnqueueWriteBuffer(devices[dev].commandQueue, devices[dev].objTrDataBase, CL_FALSE, 0, conf -> trNInstances * conf -> nFeatures * sizeof(cl_float), trDataBase, 0, NULL, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_ENQUEUE_TRDB);
181 check(clEnqueueWriteBuffer(devices[dev].commandQueue, devices[dev].objSelInstances, CL_FALSE, 0, conf -> K * sizeof(cl_int), selInstances, 0, NULL, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_ENQUEUE_CENTROIDS);
182 check(clEnqueueWriteBuffer(devices[dev].commandQueue, devices[dev].objTransposedTrDataBase, CL_FALSE, 0, conf -> trNInstances * conf -> nFeatures * sizeof(cl_float), transposedTrDataBase, 0, NULL, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_ENQUEUE_TTRDB);
183
184 // Resources used are released
185 delete[] kernelSource;
186 clReleaseProgram(program);
187
188 found = true;
189 allDevices.erase(allDevices.begin() + allDev);
190 }
191 }
192
193 check(!found, "%s\n", CL_ERROR_DEVICE_FOUND);
194 }
195
196
197 /********** Add the CPU if has been enabled in configuration ***********/
198
199 if (conf -> ompThreads > 0) {
200 devices[conf -> nDevices].deviceType = CL_DEVICE_TYPE_CPU;
201 devices[conf -> nDevices].computeUnits = conf -> ompThreads;
202 ++(conf -> nDevices);
203 }
204
205 return devices;
206}

213std::vector<cl_device_id> getAllDevices() {
214
215 // OpenCL variables
216 cl_platform_id *platforms;
217 cl_uint numPlatforms;
218 cl_uint numPlatformsDevices;
219 std::vector<cl_device_id> allDevices;
220 cl_device_id *devices;
221 cl_int status;
222
223 // Get the number of platforms
224 check(clGetPlatformIDs(0, NULL, &numPlatforms) != CL_SUCCESS, "%s\n", CL_ERROR_PLATFORMS_NUMBER);
225 check(numPlatforms == 0, "%s\n", CL_ERROR_PLATFORMS_FOUND);
226
227 // Get the platforms
228 platforms = new cl_platform_id[numPlatforms];
229 check(clGetPlatformIDs(numPlatforms, platforms, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_PLATFORM_ID);
230
231 // Search devices in each platform
232 for (int i = 0; i < numPlatforms; ++i) {
233
234 // Get the number of devices of this platform
235 status = clGetDeviceIDs(platforms[i], CL_DEVICE_TYPE_GPU, 0, 0, (cl_uint*) &numPlatformsDevices);
236 check(status != CL_SUCCESS && status != CL_DEVICE_NOT_FOUND, "%s\n", CL_ERROR_DEVICES_NUMBER);
237
238 // Get all devices of this platform
239 if (numPlatformsDevices > 0) {
240 devices = new cl_device_id[numPlatformsDevices];
241 check(clGetDeviceIDs(platforms[i], CL_DEVICE_TYPE_GPU, numPlatformsDevices, devices, NULL) != CL_SUCCESS, "%s\n", CL_ERROR_DEVICE_ID);
242 allDevices.insert(allDevices.end(), devices, devices + numPlatformsDevices);
243 delete[] devices;
244 }
245 }
246
247 delete[] platforms;
248 return allDevices;
249}
