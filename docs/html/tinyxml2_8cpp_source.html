227void StrPair::CollapseWhitespace()
228{
229 // Adjusting _start would cause undefined behavior on delete[]
230 TIXMLASSERT( ( _flags & NEEDS_DELETE ) == 0 );
231 // Trim leading space.
232 _start = XMLUtil::SkipWhiteSpace( _start );
233
234 if ( *_start ) {
235 char* p = _start; // the read pointer
236 char* q = _start; // the write pointer
237
238 while( *p ) {
239 if ( XMLUtil::IsWhiteSpace( *p )) {
240 p = XMLUtil::SkipWhiteSpace( p );
241 if ( *p == 0 ) {
242 break; // don't write to q; this trims the trailing space.
243 }
244 *q = ' ';
245 ++q;
246 }
247 *q = *p;
248 ++q;
249 ++p;
250 }
251 *q = 0;
252 }

256const char* StrPair::GetStr()
257{
258 TIXMLASSERT( _start );
259 TIXMLASSERT( _end );
260 if ( _flags & NEEDS_FLUSH ) {
261 *_end = 0;
262 _flags ^= NEEDS_FLUSH;
263
264 if ( _flags ) {
265 char* p = _start; // the read pointer
266 char* q = _start; // the write pointer
267
268 while( p < _end ) {
269 if ( (_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == CR ) {
270 // CR-LF pair becomes LF
271 // CR alone becomes LF
272 // LF-CR becomes LF
273 if ( *(p+1) == LF ) {
274 p += 2;
275 }
276 else {
277 ++p;
278 }
279 *q++ = LF;
280 }
281 else if ( (_flags & NEEDS_NEWLINE_NORMALIZATION) && *p == LF ) {
282 if ( *(p+1) == CR ) {
283 p += 2;
284 }
285 else {
286 ++p;
287 }
288 *q++ = LF;
289 }
290 else if ( (_flags & NEEDS_ENTITY_PROCESSING) && *p == '&' ) {
291 // Entities handled by tinyXML2:
292 // - special entities in the entity table [in/out]
293 // - numeric character reference [in]
294 // &#20013; or &#x4e2d;
295
296 if ( *(p+1) == '#' ) {
297 const int buflen = 10;
298 char buf[buflen] = { 0 };
299 int len = 0;
300 char* adjusted = const_cast<char*>( XMLUtil::GetCharacterRef( p, buf, &len ) );
301 if ( adjusted == 0 ) {
302 *q = *p;
303 ++p;
304 ++q;
305 }
306 else {
307 TIXMLASSERT( 0 <= len && len <= buflen );
308 TIXMLASSERT( q + len <= adjusted );
309 p = adjusted;
310 memcpy( q, buf, len );
311 q += len;
312 }
313 }
314 else {
315 bool entityFound = false;
316 for( int i = 0; i < NUM_ENTITIES; ++i ) {
317 const Entity& entity = entities[i];
318 if ( strncmp( p + 1, entity.pattern, entity.length ) == 0
319 && *( p + entity.length + 1 ) == ';' ) {
320 // Found an entity - convert.
321 *q = entity.value;
322 ++q;
323 p += entity.length + 2;
324 entityFound = true;
325 break;
326 }
327 }
328 if ( !entityFound ) {
329 // fixme: treat as error?
330 ++p;
331 ++q;
332 }
333 }
334 }
335 else {
336 *q = *p;
337 ++p;
338 ++q;
339 }
340 }
341 *q = 0;
342 }
343 // The loop below has plenty going on, and this
344 // is a less useful mode. Break it out.
345 if ( _flags & NEEDS_WHITESPACE_COLLAPSING ) {
346 CollapseWhitespace();
347 }
348 _flags = (_flags & NEEDS_DELETE);
349 }
350 TIXMLASSERT( _start );
351 return _start;
352}

377void XMLUtil::ConvertUTF32ToUTF8( unsigned long input, char* output, int* length )
378{
379 const unsigned long BYTE_MASK = 0xBF;
380 const unsigned long BYTE_MARK = 0x80;
381 const unsigned long FIRST_BYTE_MARK[7] = { 0x00, 0x00, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC };
382
383 if (input < 0x80) {
384 *length = 1;
385 }
386 else if ( input < 0x800 ) {
387 *length = 2;
388 }
389 else if ( input < 0x10000 ) {
390 *length = 3;
391 }
392 else if ( input < 0x200000 ) {
393 *length = 4;
394 }
395 else {
396 *length = 0; // This code won't convert this correctly anyway.
397 return;
398 }
399
400 output += *length;
401
402 // Scary scary fall throughs.
403 switch (*length) {
404 case 4:
405 --output;
406 *output = (char)((input | BYTE_MARK) & BYTE_MASK);
407 input >>= 6;
408 case 3:
409 --output;
410 *output = (char)((input | BYTE_MARK) & BYTE_MASK);
411 input >>= 6;
412 case 2:
413 --output;
414 *output = (char)((input | BYTE_MARK) & BYTE_MASK);
415 input >>= 6;
416 case 1:
417 --output;
418 *output = (char)(input | FIRST_BYTE_MARK[*length]);
419 break;
420 default:
421 TIXMLASSERT( false );
422 }
423}

426const char* XMLUtil::GetCharacterRef( const char* p, char* value, int* length )
427{
428 // Presume an entity, and pull it out.
429 *length = 0;
430
431 if ( *(p+1) == '#' && *(p+2) ) {
432 unsigned long ucs = 0;
433 TIXMLASSERT( sizeof( ucs ) >= 4 );
434 ptrdiff_t delta = 0;
435 unsigned mult = 1;
436 static const char SEMICOLON = ';';
437
438 if ( *(p+2) == 'x' ) {
439 // Hexadecimal.
440 const char* q = p+3;
441 if ( !(*q) ) {
442 return 0;
443 }
444
445 q = strchr( q, SEMICOLON );
446
447 if ( !q ) {
448 return 0;
449 }
450 TIXMLASSERT( *q == SEMICOLON );
451
452 delta = q-p;
453 --q;
454
455 while ( *q != 'x' ) {
456 unsigned int digit = 0;
457
458 if ( *q >= '0' && *q <= '9' ) {
459 digit = *q - '0';
460 }
461 else if ( *q >= 'a' && *q <= 'f' ) {
462 digit = *q - 'a' + 10;
463 }
464 else if ( *q >= 'A' && *q <= 'F' ) {
465 digit = *q - 'A' + 10;
466 }
467 else {
468 return 0;
469 }
470 TIXMLASSERT( digit >= 0 && digit < 16);
471 TIXMLASSERT( digit == 0 || mult <= UINT_MAX / digit );
472 const unsigned int digitScaled = mult * digit;
473 TIXMLASSERT( ucs <= ULONG_MAX - digitScaled );
474 ucs += digitScaled;
475 TIXMLASSERT( mult <= UINT_MAX / 16 );
476 mult *= 16;
477 --q;
478 }
479 }
480 else {
481 // Decimal.
482 const char* q = p+2;
483 if ( !(*q) ) {
484 return 0;
485 }
486
487 q = strchr( q, SEMICOLON );
488
489 if ( !q ) {
490 return 0;
491 }
492 TIXMLASSERT( *q == SEMICOLON );
493
494 delta = q-p;
495 --q;
496
497 while ( *q != '#' ) {
498 if ( *q >= '0' && *q <= '9' ) {
499 const unsigned int digit = *q - '0';
500 TIXMLASSERT( digit >= 0 && digit < 10);
501 TIXMLASSERT( digit == 0 || mult <= UINT_MAX / digit );
502 const unsigned int digitScaled = mult * digit;
503 TIXMLASSERT( ucs <= ULONG_MAX - digitScaled );
504 ucs += digitScaled;
505 }
506 else {
507 return 0;
508 }
509 TIXMLASSERT( mult <= UINT_MAX / 10 );
510 mult *= 10;
511 --q;
512 }
513 }
514 // convert the UCS to UTF-8
515 ConvertUTF32ToUTF8( ucs, value, length );
516 return p + delta + 1;
517 }
518 return p+1;
519}

608{

610 TIXMLASSERT( p );

619 // These strings define the matching patterns:
620 static const char* xmlHeader = { "<?" };
621 static const char* commentHeader = { "<!--" };
622 static const char* cdataHeader = { "<![CDATA[" };
623 static const char* dtdHeader = { "<!" };
624 static const char* elementHeader = { "<" }; // and a header for everything else; check last.
625
626 static const int xmlHeaderLen = 2;
627 static const int commentHeaderLen = 4;
628 static const int cdataHeaderLen = 9;
629 static const int dtdHeaderLen = 2;
630 static const int elementHeaderLen = 1;
631
632 TIXMLASSERT( sizeof( XMLComment ) == sizeof( XMLUnknown ) ); // use same memory pool
633 TIXMLASSERT( sizeof( XMLComment ) == sizeof( XMLDeclaration ) ); // use same memory pool
634 XMLNode* returnNode = 0;
635 if ( XMLUtil::StringEqual( p, xmlHeader, xmlHeaderLen ) ) {
636 TIXMLASSERT( sizeof( XMLDeclaration ) == _commentPool.ItemSize() );
637 returnNode = new (_commentPool.Alloc()) XMLDeclaration( this );
638 returnNode->_memPool = &_commentPool;
639 p += xmlHeaderLen;
640 }
641 else if ( XMLUtil::StringEqual( p, commentHeader, commentHeaderLen ) ) {
642 TIXMLASSERT( sizeof( XMLComment ) == _commentPool.ItemSize() );
643 returnNode = new (_commentPool.Alloc()) XMLComment( this );
644 returnNode->_memPool = &_commentPool;
645 p += commentHeaderLen;
646 }
647 else if ( XMLUtil::StringEqual( p, cdataHeader, cdataHeaderLen ) ) {
648 TIXMLASSERT( sizeof( XMLText ) == _textPool.ItemSize() );
649 XMLText* text = new (_textPool.Alloc()) XMLText( this );
650 returnNode = text;
651 returnNode->_memPool = &_textPool;
652 p += cdataHeaderLen;
653 text->SetCData( true );
654 }
655 else if ( XMLUtil::StringEqual( p, dtdHeader, dtdHeaderLen ) ) {
656 TIXMLASSERT( sizeof( XMLUnknown ) == _commentPool.ItemSize() );
657 returnNode = new (_commentPool.Alloc()) XMLUnknown( this );
658 returnNode->_memPool = &_commentPool;
659 p += dtdHeaderLen;
660 }
661 else if ( XMLUtil::StringEqual( p, elementHeader, elementHeaderLen ) ) {
662 TIXMLASSERT( sizeof( XMLElement ) == _elementPool.ItemSize() );
663 returnNode = new (_elementPool.Alloc()) XMLElement( this );
664 returnNode->_memPool = &_elementPool;
665 p += elementHeaderLen;
666 }
667 else {
668 TIXMLASSERT( sizeof( XMLText ) == _textPool.ItemSize() );
669 returnNode = new (_textPool.Alloc()) XMLText( this );
670 returnNode->_memPool = &_textPool;
671 p = start; // Back it up, all the text counts.
672 }
673
674 TIXMLASSERT( returnNode );
675 TIXMLASSERT( p );
676 *node = returnNode;
677 return p;
678}

925char* XMLNode::ParseDeep( char* p, StrPair* parentEnd )
926{
927 // This is a recursive method, but thinking about it "at the current level"
928 // it is a pretty simple flat list:
929 // <foo/>
930 // <!-- comment -->
931 //
932 // With a special case:
933 // <foo>
934 // </foo>
935 // <!-- comment -->
936 //
937 // Where the closing element (/foo) *must* be the next thing after the opening
938 // element, and the names must match. BUT the tricky bit is that the closing
939 // element will be read by the child.
940 //
941 // 'endTag' is the end tag for this node, it is returned by a call to a child.
942 // 'parentEnd' is the end tag for the parent, which is filled in and returned.
943
944 while( p && *p ) {
945 XMLNode* node = 0;
946
947 p = _document->Identify( p, &node );
948 if ( node == 0 ) {
949 break;
950 }
951
952 StrPair endTag;
953 p = node->ParseDeep( p, &endTag );
954 if ( !p ) {
955 DeleteNode( node );
956 if ( !_document->Error() ) {
957 _document->SetError( XML_ERROR_PARSING, 0, 0 );
958 }
959 break;
960 }
961
962 XMLDeclaration* decl = node->ToDeclaration();
963 if ( decl ) {
964 // A declaration can only be the first child of a document.
965 // Set error, if document already has children.
966 if ( !_document->NoChildren() ) {
967 _document->SetError( XML_ERROR_PARSING_DECLARATION, decl->Value(), 0);
968 DeleteNode( decl );
969 break;
970 }
971 }
972
973 XMLElement* ele = node->ToElement();
974 if ( ele ) {
975 // We read the end tag. Return it to the parent.
976 if ( ele->ClosingType() == XMLElement::CLOSING ) {
977 if ( parentEnd ) {
978 ele->_value.TransferTo( parentEnd );
979 }
980 node->_memPool->SetTracked(); // created and then immediately deleted.
981 DeleteNode( node );
982 return p;
983 }
984
985 // Handle an end tag returned to this level.
986 // And handle a bunch of annoying errors.
987 bool mismatch = false;
988 if ( endTag.Empty() ) {
989 if ( ele->ClosingType() == XMLElement::OPEN ) {
990 mismatch = true;
991 }
992 }
993 else {
994 if ( ele->ClosingType() != XMLElement::OPEN ) {
995 mismatch = true;
996 }
997 else if ( !XMLUtil::StringEqual( endTag.GetStr(), node->Value() ) ) {
998 mismatch = true;
999 }
1000 }
1001 if ( mismatch ) {
1002 _document->SetError( XML_ERROR_MISMATCHED_ELEMENT, node->Value(), 0 );
1003 DeleteNode( node );
1004 break;
1005 }
1006 }
1007 InsertEndChild( node );
1008 }
1009 return 0;
1010}

1584
1585char* XMLElement::ParseAttributes( char* p )
1586{
1587 const char* start = p;
1588 XMLAttribute* prevAttribute = 0;
1589
1590 // Read the attributes.
1591 while( p ) {
1592 p = XMLUtil::SkipWhiteSpace( p );
1593 if ( !(*p) ) {
1594 _document->SetError( XML_ERROR_PARSING_ELEMENT, start, Name() );
1595 return 0;
1596 }
1597
1598 // attribute.
1599 if (XMLUtil::IsNameStartChar( *p ) ) {
1600 TIXMLASSERT( sizeof( XMLAttribute ) == _document->_attributePool.ItemSize() );
1601 XMLAttribute* attrib = new (_document->_attributePool.Alloc() ) XMLAttribute();
1602 attrib->_memPool = &_document->_attributePool;
1603 attrib->_memPool->SetTracked();
1604
1605 p = attrib->ParseDeep( p, _document->ProcessEntities() );
1606 if ( !p || Attribute( attrib->Name() ) ) {
1607 DeleteAttribute( attrib );
1608 _document->SetError( XML_ERROR_PARSING_ATTRIBUTE, start, p );
1609 return 0;
1610 }
1611 // There is a minor bug here: if the attribute in the source xml
1612 // document is duplicated, it will not be detected and the
1613 // attribute will be doubly added. However, tracking the 'prevAttribute'
1614 // avoids re-scanning the attribute list. Preferring performance for
1615 // now, may reconsider in the future.
1616 if ( prevAttribute ) {
1617 prevAttribute->_next = attrib;
1618 }
1619 else {
1620 _rootAttribute = attrib;
1621 }
1622 prevAttribute = attrib;
1623 }
1624 // end of the tag
1625 else if ( *p == '>' ) {
1626 ++p;
1627 break;
1628 }
1629 // end of the tag
1630 else if ( *p == '/' && *(p+1) == '>' ) {
1631 _closingType = CLOSED;
1632 return p+2; // done; sealed element.
1633 }
1634 else {
1635 _document->SetError( XML_ERROR_PARSING_ELEMENT, start, p );
1636 return 0;
1637 }
1638 }
1639 return p;
1640}
