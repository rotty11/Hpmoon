101int* getPool(const Config *const conf) {
102
103 // Create and fill the pool
104 int *pool = new int[conf -> poolSize];
105 std::set<int> candidates;
106 for (int i = 0; i < conf -> poolSize; ++i) {
107
108 // std::set guarantees unique elements in the insert function
109 for (int j = 0; j < conf -> tourSize; ++j) {
110 candidates.insert(rand() % conf -> subpopulationSize);
111 }
112
113 // At this point, the individuals already are sorted by rank and crowding distance
114 // Therefore, lower index is better
115 pool[i] = *(candidates.begin());
116 candidates.clear();
117 }

130int crossoverUniform(Individual *const subpop, const int *const pool, const Config *const conf) {
131
132 // Reset the children
133 for (int i = conf -> subpopulationSize; i < conf -> familySize; ++i) {
134 memset(subpop[i].chromosome, 0, conf -> nFeatures * sizeof(unsigned char));
135 for (unsigned char obj = 0; obj < conf -> nObjectives; ++obj) {
136 subpop[i].fitness[obj] = 0.0f;
137 }
138 subpop[i].nSelFeatures = 0;
139 subpop[i].rank = -1;
140 subpop[i].crowding = 0.0f;
141 }
142
143 Individual *child = &(subpop[conf -> subpopulationSize]);
144 for (int i = 0; i < conf -> poolSize; ++i) {
145
146 // 75% probability perform crossover. Two childen are generated
147 Individual *parent1 = &(subpop[pool[rand() % conf -> poolSize]]);
148 if ((rand() / (float) RAND_MAX) < 0.75f) {
149
150 // Avoid repeated parents
151 Individual *parent2 = &(subpop[pool[rand() % conf -> poolSize]]);
152 Individual *child2 = child + 1;
153 while (parent1 == parent2) {
154 parent2 = &(subpop[pool[rand() % conf -> poolSize]]);
155 }
156
157 // Perform uniform crossover for each decision variable in the chromosome
158 for (int f = 0; f < conf -> nFeatures; ++f) {
159
160 // 50% probability perform copy the decision variable of the other parent
161 if ((parent1 -> chromosome[f] != parent2 -> chromosome[f]) && ((rand() / (float) RAND_MAX) < 0.5f)) {
162 child -> nSelFeatures += (child -> chromosome[f] = parent2 -> chromosome[f]);
163 child2 -> nSelFeatures += (child2 -> chromosome[f] = parent1 -> chromosome[f]);
164 }
165 else {
166 child -> nSelFeatures += (child -> chromosome[f] = parent1 -> chromosome[f]);
167 child2 -> nSelFeatures += (child2 -> chromosome[f] = parent2 -> chromosome[f]);
168 }
169 }
170
171 // At least one decision variable must be set to '1'
172 if (child -> nSelFeatures == 0) {
173 child -> chromosome[rand() % conf -> nFeatures] = child -> nSelFeatures = 1;
174 }
175
176 if (child2 -> nSelFeatures == 0) {
177 child2 -> chromosome[rand() % conf -> nFeatures] = child2 -> nSelFeatures = 1;
178 }
179 child += 2;
180 }
181
182 // 25% probability perform mutation. One child is generated
183 // Mutation is based on random mutation
184 else {
185
186 // Perform mutation on each element of the selected parent
187 for (int f = 0; f < conf -> nFeatures; ++f) {
188
189 // 10% probability perform mutation (gen level)
190 float probability = (rand() / (float) RAND_MAX);
191 if (probability < 0.1f) {
192 if ((rand() / (float) RAND_MAX) > 0.01f) {
193 child -> chromosome[f] = 0;
194 }
195 else {
196 child -> nSelFeatures += (child -> chromosome[f] = 1);
197 }
198 }
199 else {
200 child -> nSelFeatures += (child -> chromosome[f] = parent1 -> chromosome[f]);
201 }
202 }
203
204 // At least one decision variable must be set to '1'
205 if (child -> nSelFeatures == 0) {
206 child -> chromosome[rand() % conf -> nFeatures] = child -> nSelFeatures = 1;
207 }
208 ++child;
209 }
210 }
211
212 return child - (subpop + conf -> subpopulationSize);
213}

223void migration(Individual *const subpops, const int nSubpopulations, const int *const nIndsFronts0, const Config *const conf) {
224
225 // From subpopulations randomly choosen some individuals of the front 0 are copied to each subpopulation (the worst individuals are deleted)
226 for (int subpop = 0; subpop < nSubpopulations; ++subpop) {
227
228 // This vector contains the available subpopulations indexes which are randomly choosen for copy the individuals of the front 0
229 std::vector<int> randomIndex(nSubpopulations);
230 std::iota(randomIndex.begin(), randomIndex.end(), 0);
231
232 // The current subpopulation will not copy its own individuals
233 randomIndex.erase(randomIndex.begin() + subpop);
234 std::random_shuffle(randomIndex.begin(), randomIndex.end());
235
236 int maxCopy = conf -> subpopulationSize - nIndsFronts0[subpop];
237 Individual *ptrDest = subpops + (subpop * conf -> familySize) + conf -> subpopulationSize;
238 for (int subpop2 = 0; subpop2 < nSubpopulations - 1 && maxCopy > 0; ++subpop2) {
239 int toCopy = std::min(maxCopy, nIndsFronts0[randomIndex[subpop2]] >> 1);
240 Individual *ptrOrig = subpops + (randomIndex[subpop2] * conf -> familySize);
241 ptrDest -= toCopy;
242 memcpy(ptrDest, ptrOrig, toCopy * sizeof(Individual));
243 maxCopy -= toCopy;
244 }
245 }
246
247 #pragma omp parallel for
248 for (int sp = 0; sp < nSubpopulations; ++sp) {
249 int popIndex = sp * conf -> familySize;
250
251 // The crowding distance of the subpopulation is initialized again for the next nonDominationSort
252 for (int i = popIndex; i < popIndex + conf -> subpopulationSize; ++i) {
253 subpops[i].crowding = 0.0f;
254 }
255 nonDominationSort(subpops + popIndex, conf -> subpopulationSize, conf);
256 }

270void evolve(Individual *const subpop, int *const nIndsFronts0, CLDevice *const devicesObject, const float *const trDataBase, const int *const selInstances, const Config *const conf, const bool initialize) {

274
275 int nDevices = (omp_get_num_threads() > 1) ? 1 : conf -> nDevices;
276 if (initialize) {
277 evaluation(subpop, conf -> subpopulationSize, devicesObject, nDevices, trDataBase, selInstances, conf);
278
279
280 /********** Sort the subpopulation with the 'Non-dominated sorting' method ***********/
281
282 nIndsFronts0[0] = nonDominationSort(subpop, conf -> subpopulationSize, conf);
283 }
284
285
286 /********** Start the evolution process ***********/
287
288 for (int g = 0; g < conf -> nGenerations; ++g) {
289
290
291 /********** Fill the mating pool and perform crossover ***********/
292
293 const int *const pool = getPool(conf);
294 int nChildren = crossoverUniform(subpop, pool, conf);
295
296 // Local resources used are released
297 delete[] pool;
298
299
300 /********** Multi-objective individuals evaluation over the subpopulation ***********/
301
302 evaluation(subpop + conf -> subpopulationSize, nChildren, devicesObject, nDevices, trDataBase, selInstances, conf);
303
304
305 /********** The crowding distance of the parents is initialized again for the next nonDominationSort ***********/
306
307 for (int i = 0; i < conf -> subpopulationSize; ++i) {
308 subpop[i].crowding = 0.0f;
309 }
310
311 // Replace subpopulation
312 // Parents and children are sorted by rank and crowding distance.
313 // The first 'conf -> subpopulationSize' individuals will continue for the next generation
314 nIndsFronts0[0] = nonDominationSort(subpop, conf -> subpopulationSize + nChildren, conf);
315 }

327void agIslands(Individual *subpops, CLDevice *const devicesObject, const float *const trDataBase, const int *const selInstances, const Config *const conf) {
328
329
330 /********** MPI variables ***********/
331
332 MPI::Status status;
333 int array_of_blocklengths[3] = {conf -> nFeatures, conf -> nObjectives + 1, 2};
334 MPI::Datatype array_of_types[3] = {MPI::UNSIGNED_CHAR, MPI::FLOAT, MPI::INT};
335
336 // The 'Individual' datatype must be converted to a MPI datatype and commit it
337 MPI::Aint array_of_displacement[3] = {offsetof(Individual, chromosome), offsetof(Individual, fitness), offsetof(Individual, rank)};
338 MPI::Datatype Individual_MPI_type = MPI::Datatype::Create_struct(3, array_of_blocklengths, array_of_displacement, array_of_types);
339 Individual_MPI_type.Commit();
340
341
342 /******* Measure and start the master-worker algorithm *******/
343
344 MPI::COMM_WORLD.Barrier();
345
346 // Master
347 if (conf -> mpiRank == 0) {
348 double timeStart = omp_get_wtime();
349 MPI::Request requests[conf -> mpiSize - 1];
350 int nIndsFronts0[conf -> nSubpopulations];
351 int finalFront0;
352
353 // I work alone
354 if (conf -> mpiSize == 1) {
355 omp_set_nested(1);
356 int nThreads = std::min(conf -> nDevices, conf -> nSubpopulations);
357 for (int gMig = 0; gMig < conf -> nGlobalMigrations; ++gMig) {
358
359 #pragma omp parallel for num_threads(nThreads) schedule(dynamic, 1)
360 for (int sp = 0; sp < conf -> nSubpopulations; ++sp) {
361 int popIndex = sp * conf -> familySize;
362 evolve(subpops + popIndex, &nIndsFronts0[sp], &devicesObject[omp_get_thread_num()], trDataBase, selInstances, conf, gMig == 0);
363 }
364
365 // Migration process between subpopulations
366 if (gMig != conf -> nGlobalMigrations - 1 && conf -> nSubpopulations > 1) {
367 migration(subpops, conf -> nSubpopulations, nIndsFronts0, conf);
368 }
369 }
370 }
371
372 // I need to distribute
373 else {
374
375 // The master receives the number of subpopulations that each worker can process
376 int workerCapacities[conf -> mpiSize - 1];
377 for (int p = 1; p < conf -> mpiSize; ++p) {
378 requests[p - 1] = MPI::COMM_WORLD.Irecv(&workerCapacities[p - 1], 1, MPI::INT, p, MPI::ANY_TAG);
379 }
380
381 MPI::Request::Waitall(conf -> mpiSize - 1, requests);
382
383
384 /********** In each migration the individuals are exchanged between subpopulations of different nodes ***********/
385
386 for (int gMig = 0; gMig < conf -> nGlobalMigrations; ++gMig) {
387
388 // Send some work to the workers
389 int nextWork = 0;
390 int sent = 0;
391 int mpiTag = (gMig == 0) ? INITIALIZE : IGNORE_VALUE;
392 for (int p = 1; p < conf -> mpiSize && nextWork < conf -> nSubpopulations; ++p) {
393 int finallyWork = std::min(workerCapacities[p - 1], conf -> nSubpopulations - nextWork);
394 int popIndex = nextWork * conf -> familySize;
395 requests[p - 1] = MPI::COMM_WORLD.Isend(subpops + popIndex, finallyWork * conf -> familySize, Individual_MPI_type, p, mpiTag);
396 nextWork += finallyWork;
397 ++sent;
398 }
399 MPI::Request::Waitall(sent, requests);
400
401 // Dynamically distribute the subpopulations
402 int receivedPtr = 0;
403 while (nextWork < conf -> nSubpopulations) {
404 MPI::COMM_WORLD.Recv(subpops + (receivedPtr * conf -> familySize), conf -> familySize, Individual_MPI_type, MPI::ANY_SOURCE, MPI::ANY_TAG, status);
405 int popIndex = nextWork * conf -> familySize;
406 MPI::COMM_WORLD.Send(subpops + popIndex, conf -> familySize, Individual_MPI_type, status.Get_source(), mpiTag);
407 nIndsFronts0[receivedPtr] = status.Get_tag();
408 ++receivedPtr;
409 ++nextWork;
410 }
411
412 // Receive the remaining work
413 while (receivedPtr < conf -> nSubpopulations) {
414 MPI::COMM_WORLD.Recv(subpops + (receivedPtr * conf -> familySize), conf -> familySize, Individual_MPI_type, MPI::ANY_SOURCE, MPI::ANY_TAG, status);
415 MPI::COMM_WORLD.Send(NULL, 0, MPI::INT, status.Get_source(), FINISH);
416 nIndsFronts0[receivedPtr] = status.Get_tag();
417 ++receivedPtr;
418 }
419
420 // Migration process between subpopulations of different nodes
421 if (gMig != conf -> nGlobalMigrations - 1 && conf -> nSubpopulations > 1) {
422 migration(subpops, conf -> nSubpopulations, nIndsFronts0, conf);
423 }
424 }
425
426 // Notify to all workers that the work has finished
427 for (int p = 1; p < conf -> mpiSize; ++p) {
428 requests[p - 1] = MPI::COMM_WORLD.Isend(NULL, 0, MPI::INT, p, FINISH);
429 }
430 }
431
432
433 /********** Recombination process ***********/
434
435 if (conf -> nSubpopulations > 1) {
436 for (int sp = 0; sp < conf -> nSubpopulations; ++sp) {
437 memcpy(subpops + (sp * conf -> subpopulationSize), subpops + (sp * conf -> familySize), conf -> subpopulationSize * sizeof(Individual));
438 }
439
440 // The crowding distance of the subpopulation is initialized again for the next nonDominationSort
441 #pragma omp parallel for
442 for (int i = 0; i < conf -> worldSize; ++i) {
443 subpops[i].crowding = 0.0f;
444 }
445 finalFront0 = std::min(conf -> subpopulationSize, nonDominationSort(subpops, conf -> worldSize, conf));
446 }
447 else {
448 finalFront0 = nIndsFronts0[0];
449 }
450
451 // All processes must reach this point in order to provide a real time measure
452 MPI::Request::Waitall(conf -> mpiSize - 1, requests);
453 MPI::COMM_WORLD.Barrier();
454 fprintf(stdout, "%.10g\n", (omp_get_wtime() - timeStart) * 1000.0);
455
456 // Get the hypervolume
457 fprintf(stdout, "%.6g\n", getHypervolume(subpops, finalFront0, conf));
458
459 // Generation of the Gnuplot file for display the Pareto front
460 generateDataPlot(subpops, finalFront0, conf);
461 generateGnuplot(conf);
462 }
463
464 // Workers
465 else {
466
467 // The worker tells to the master how many subpopulations can be processed
468 MPI::COMM_WORLD.Isend(&(conf -> nDevices), 1, MPI::INT, 0, 0);
469 omp_set_nested(1);
470 subpops = new Individual[conf -> nDevices * conf -> familySize];
471
472 // The worker receives as many subpopulations as number of devices at most
473 MPI::COMM_WORLD.Recv(subpops, conf -> nDevices * conf -> familySize, Individual_MPI_type, 0, MPI::ANY_TAG, status);
474
475 while (status.Get_tag() != FINISH) {
476 int nSubpopulations = status.Get_count(Individual_MPI_type) / conf -> familySize;
477 int EXIT = false;
478
479 #pragma omp parallel num_threads(nSubpopulations)
480 {
481 int threadID = omp_get_thread_num();
482 MPI::Request request;
483 MPI::Status stat = status;
484 int nIndsFronts0;
485 int popIndex = threadID * conf -> familySize;
486 do {
487 evolve(subpops + popIndex, &nIndsFronts0, &devicesObject[threadID], trDataBase, selInstances, conf, stat.Get_tag() == INITIALIZE);
488
489 // The Worker sends to the master the subpopulations already evaluated and will request new work
490 request = MPI::COMM_WORLD.Isend(subpops + popIndex, conf -> familySize, Individual_MPI_type, 0, nIndsFronts0);
491 request.Wait();
492 MPI::COMM_WORLD.Recv(subpops + popIndex, conf -> familySize, Individual_MPI_type, 0, MPI::ANY_TAG, stat);
493 } while (stat.Get_tag() != FINISH);
494 }
495
496 MPI::COMM_WORLD.Recv(subpops, conf -> nDevices * conf -> familySize, Individual_MPI_type, 0, MPI::ANY_TAG, status);
497 }
498
499 // All process must reach this point in order to provide a real time measure
500 MPI::COMM_WORLD.Barrier();
501 }
502
503 // Variables used by both master and workers are released
504 Individual_MPI_type.Free();
505}
