130 int crossoverUniform(Individual *const subpop, const int *const pool, const Config *const conf) {
131 
132  // Reset the children
133  for (int i = conf -> subpopulationSize; i < conf -> familySize; ++i) {
134  memset(subpop[i].chromosome, 0, conf -> nFeatures * sizeof(unsigned char));
135  for (unsigned char obj = 0; obj < conf -> nObjectives; ++obj) {
136  subpop[i].fitness[obj] = 0.0f;
137  }
138  subpop[i].nSelFeatures = 0;
139  subpop[i].rank = -1;
140  subpop[i].crowding = 0.0f;
141  }
142 
143  Individual *child = &(subpop[conf -> subpopulationSize]);
144  for (int i = 0; i < conf -> poolSize; ++i) {
145 
146  // 75% probability perform crossover. Two childen are generated
147  Individual *parent1 = &(subpop[pool[rand() % conf -> poolSize]]);
148  if ((rand() / (float) RAND_MAX) < 0.75f) {
149 
150  // Avoid repeated parents
151  Individual *parent2 = &(subpop[pool[rand() % conf -> poolSize]]);
152  Individual *child2 = child + 1;
153  while (parent1 == parent2) {
154  parent2 = &(subpop[pool[rand() % conf -> poolSize]]);
155  }
156 
157  // Perform uniform crossover for each decision variable in the chromosome
158  for (int f = 0; f < conf -> nFeatures; ++f) {
159 
160  // 50% probability perform copy the decision variable of the other parent
161  if ((parent1 -> chromosome[f] != parent2 -> chromosome[f]) && ((rand() / (float) RAND_MAX) < 0.5f)) {
162  child -> nSelFeatures += (child -> chromosome[f] = parent2 -> chromosome[f]);
163  child2 -> nSelFeatures += (child2 -> chromosome[f] = parent1 -> chromosome[f]);
164  }
165  else {
166  child -> nSelFeatures += (child -> chromosome[f] = parent1 -> chromosome[f]);
167  child2 -> nSelFeatures += (child2 -> chromosome[f] = parent2 -> chromosome[f]);
168  }
169  }
170 
171  // At least one decision variable must be set to '1'
172  if (child -> nSelFeatures == 0) {
173  child -> chromosome[rand() % conf -> nFeatures] = child -> nSelFeatures = 1;
174  }
175 
176  if (child2 -> nSelFeatures == 0) {
177  child2 -> chromosome[rand() % conf -> nFeatures] = child2 -> nSelFeatures = 1;
178  }
179  child += 2;
180  }
181 
182  // 25% probability perform mutation. One child is generated
183  // Mutation is based on random mutation
184  else {
185 
186  // Perform mutation on each element of the selected parent
187  for (int f = 0; f < conf -> nFeatures; ++f) {
188 
189  // 10% probability perform mutation (gen level)
190  float probability = (rand() / (float) RAND_MAX);
191  if (probability < 0.1f) {
192  if ((rand() / (float) RAND_MAX) > 0.01f) {
193  child -> chromosome[f] = 0;
194  }
195  else {
196  child -> nSelFeatures += (child -> chromosome[f] = 1);
197  }
198  }
199  else {
200  child -> nSelFeatures += (child -> chromosome[f] = parent1 -> chromosome[f]);
201  }
202  }
203 
204  // At least one decision variable must be set to '1'
205  if (child -> nSelFeatures == 0) {
206  child -> chromosome[rand() % conf -> nFeatures] = child -> nSelFeatures = 1;
207  }
208  ++child;
209  }
210  }
211 
212  return child - (subpop + conf -> subpopulationSize);
213 }
